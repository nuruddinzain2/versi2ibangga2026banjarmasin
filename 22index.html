function saveEmailToSheet(params) { try { var ss = SpreadsheetApp.getActiveSpreadsheet(); var sheet = ss.getSheetByName("Email"); if (!sheet) { sheet = ss.insertSheet("Email"); sheet.getRange(1, 1, 1, 4).setValues([["No", "Nama", "Kode", "Email"]]); sheet.getRange(1, 1, 1, 4).setFontWeight("bold"); } var nama = params.nama || ""; var kode = params.kode || ""; var email = params.email || ""; var rowId = params.rowId; // CEK DUPLIKASI EMAIL (kecuali sedang edit data sendiri) var data = sheet.getDataRange().getValues(); for (var i = 1; i < data.length; i++) { var existingEmail = data[i][3]?.toString().trim().toLowerCase(); var existingRowId = i + 1; // Jika email sudah ada DAN bukan sedang edit baris yang sama if (existingEmail === email.trim().toLowerCase() && existingRowId != rowId) { return { error: "Email '" + email + "' sudah terdaftar oleh petugas lain!" }; } } // CEK DUPLIKASI KODE PETUGAS for (var i = 1; i < data.length; i++) { var existingKode = data[i][2]?.toString().trim(); var existingRowId = i + 1; if (existingKode === kode.trim() && existingRowId != rowId) { return { error: "Kode petugas '" + kode + "' sudah digunakan!" }; } } // Lanjut simpan atau update if (rowId) { sheet.getRange(parseInt(rowId), 2).setValue(nama); sheet.getRange(parseInt(rowId), 3).setValue(kode); sheet.getRange(parseInt(rowId), 4).setValue(email); return { message: "Data berhasil diupdate!" }; } else { var lastRow = sheet.getLastRow(); sheet.getRange(lastRow + 1, 1).setValue(lastRow); sheet.getRange(lastRow + 1, 2).setValue(nama); sheet.getRange(lastRow + 1, 3).setValue(kode); sheet.getRange(lastRow + 1, 4).setValue(email); return { message: "Data berhasil ditambahkan!" }; } } catch(e) { return { error: e.toString() }; } }
